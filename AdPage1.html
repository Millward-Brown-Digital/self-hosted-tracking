This page serves up ad #1, for campaign 12345, and records an impression record on exposure. 

                            
(function () {
    // this value to be provided by MBD
    var campaignId = 12345;

    //init MBD tracking with all default settings
    var mbd = new MBD();

    //record an impression for this creative
    //can pass anything you need to track here
    mbd.RecordImpression(campaignId, {
        creativeId: 'fear1234',
        placementId: 'placement1234',
        actionID: 1
    });
})();
